class Solution {
public:
    bool isValidSudoku(vector<vector<char>>& board) {
        for(int i = 0;i<9;i++)
        {
            int freq[10] = {0};
            for(int j = 0;j<9;j++)
            {   
                if(board[i][j] == '.')
                continue;
                freq[board[i][j] - '0']++;
            }
            for(int k = 0;k<10;k++)
            {
                if(freq[k] > 1)
                {
                    return false;
                }
            }
        }


        for(int i = 0;i<9;i++)
        {   
            int freq[10] = {0};
            for(int j = 0;j<9;j++)
            {   
                if(board[j][i] == '.')
                continue;
                freq[board[j][i] - '0']++;
            }
            for(int k = 0;k<10;k++)
            {
                if(freq[k] > 1)
                {
                    return false;
                }
            }
        }

        for(int boxRow = 0; boxRow < 3; boxRow++) {
            for(int boxCol = 0; boxCol < 3; boxCol++) {

                int freq[10] = {0};

                for(int i = 0; i < 3; i++) {
                    for(int j = 0; j < 3; j++) {

                        int row = boxRow * 3 + i;
                        int col = boxCol * 3 + j;

                        if(board[row][col] == '.') continue;

                        int num = board[row][col] - '0';
                        if(++freq[num] > 1)
                            return false;
                    }
                }
            }
        }
        return true;
    }
};
