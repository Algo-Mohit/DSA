class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int size = nums.size();
        unordered_map<int ,int > count;
        for(int i =0;i<size;i++)
        {
            count[nums[i]]++;
        }
        for(auto i:count)
        {
            if(i.second > size/2)
            {
                return i.first;
            }
        }
    }
};
