class Solution {
public:
    vector<int> majorityElement(vector<int>& nums) {
        unordered_map<int,int> freq;
        vector<int> answer;
        for(int i:nums)
        {
            freq[i]++;
        }

        for(auto num : freq)
        {
            if(num.second >nums.size()/3)
            {
                answer.push_back(num.first);
            }
        }
        return answer;
    }
};
