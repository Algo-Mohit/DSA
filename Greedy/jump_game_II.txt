class Solution {
public:
    int jump(vector<int>& nums) {
        int jumps = 0 , l = 0, r = 0;
        if(nums.size()==1)return 0;
        while(r < nums.size()-1)
        {
            int far_index = 0;
            for(int i = l ;i<=r;i++)
            {
             far_index = max(i+nums[i],far_index);
            }
            l = r + 1;
            r = far_index;
            jumps += 1;
        }
        return jumps;
    }
};
