class Solution {
public:
    long long minEnd(int n, int x) {
        long long ans = x;
        n--;  
        long long bit = 1;

        while(n){
            if((x & bit) == 0){
                if(n & 1){
                    ans |= bit;
                }
                n >>= 1;
            }
            bit <<= 1;
        }

        return ans;
    }
};
