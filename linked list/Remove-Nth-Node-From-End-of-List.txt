class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        ListNode* a = new ListNode(0);
        a->next = head;
        ListNode* b = a;
        ListNode* c = a;
        for(int i = 0; i < n; i++) {
            b = b->next;
        }
        while(b->next != NULL) 
        {
            b = b->next;
            c = c->next;
        }
        c->next = c->next->next;
        return a->next;
    }
};
