class Solution {
public:
    bool hasCycle(ListNode *head) {
        if (!head) return false;

        ListNode *a = head;   
        ListNode *b = head;   

        while (b && b->next) {
            a = a->next;        
            b = b->next->next;  

            if (a == b)
                return true;    
        }
        return false;           
    }
};
