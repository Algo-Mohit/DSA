class Solution {
public:
    int maxProfit(vector<int>& prices) {
      int l = 0;
      int r = l+1;
      int maximum = 0;
      
      while(r!=prices.size())
      {
        maximum = max(maximum,prices[r]-prices[l]);
        if(prices[r]>prices[l])
        {
            r++;
        }
        else
        {
           l = r;
           r++;
        }
        
      }
     return maximum;
    }
};
